Conditional logic

.Show "description" parameter if set. If not set, show "summary" parameter.
[source,Go]
----
{{ if (isset .Params "description") }}
    {{ index .Params "description" }}
{{ else if (isset .Params "summary") }}
    {{ index .Params "summary" }}
{{ else }}
    {{ .Summary }}
{{ end }}
----
